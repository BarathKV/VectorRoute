 PART 11: KEY FUNCTION SIGNATURES AND CALL PATTERNS
═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════════

  Agent.__init__(model, db=None, tracker=None)
       ├─► FileTracker() or tracker
       ├─► DBConnection() or db
       ├─► tracker.get_file_changes() → Dict of changes
       ├─► db.update_db(changes=changes)
       └─► tracker.get_tool_registry() → Dict[str, callable]


  Agent.run(user_input: str) → Tuple[response, List[str]]
       ├─► AskSession(user_input, model)
       ├─► db.route_query(user_input) → List[str] (tool names)
       ├─► DBConnection._load_tool_docs_map() → Dict[str, tuple]
       ├─► ollama_chat(model, messages, tools) → response
       ├─► Extract tool_calls from response
       ├─► tool_registry[tool_name](**arguments) → result
       ├─► ollama_chat(...with result...) → final_response
       └─► OllamaLogger.log_call(...) → SQLite insert


  Agent.ask(user_input: str) → Tuple[response, List[str]]
       ├─► AskSession(user_input, model)
       ├─► QueryDecomposer(model).decompose(user_input, messages)
       │         ├─► ollama_chat(model, system_prompt, user_input)
       │         ├─► _extract_json(response_content) → dict
       │         └─► ExecutionPlan([Task, Task, ...])
       ├─► TaskExecutor(model).execute(plan, db, tool_registry)
       │         ├─► For each Task in plan.tasks:
       │         │    ├─► resolve_placeholders(task.query)
       │         │    ├─► task.run(db, model, tool_registry)
       │         │    │    ├─► db.route_query(resolved_query)
       │         │    │    ├─► DBConnection._load_tool_docs_map()
       │         │    │    ├─► ollama_chat(model, messages, tools)
       │         │    │    ├─► Extract & execute tool_calls
       │         │    │    └─► Return (result, tools_used)
       │         │    └─► Mark task.status = "completed"
       │         └─► Return aggregated (final_response, all_tools_used)
       └─► Return final_message, tools_used


  DBConnection.route_query(user_input: str) → List[str]
       ├─► embedding.get_embedding(user_input) → Vector
       ├─► self.collection.query(query_embeddings=[vector], n_results=K)
       └─► Return list of matching tool names


  DatabaseConnection.update_db(changes: Dict)
       ├─► For each tool in changes["new_tools"]:
       │    ├─► _load_tool_docs_map() → get tool data
       │    └─► _add_tool(tool_name, tool_data)
       │         ├─► Load capability JSON
       │         ├─► For each facet (description, domain, examples):
       │         │    ├─► embedding.get_embedding(facet_text)
       │         │    └─► collection.add(...)
       │         └─► Update tools table in SQLite
       └─► Similar for modified_tools and deleted_tools


  FileTracker.get_file_changes() → Dict[str, set]
       ├─► Scan VectorRoute-Tools/capabilities/*.json
       ├─► Scan VectorRoute-Tools/functions/*.py
       ├─► For each file:
       │    ├─► _compute_file_hash(file_path) → SHA256
       │    ├─► Query SQLite tools table for cached hash
       │    ├─► Compare hashes
       │    └─► Categorize as new/modified/deleted/unchanged
       └─► Return {"new_tools": set, "modified_tools": set, ...}


  FileTracker.get_tool_registry() → Dict[str, callable]
       ├─► Load all .py files from functions/ folder
       ├─► Dynamically import each module
       ├─► Extract callable functions
       └─► Return {tool_name: function_reference}


  ollama_chat(model, messages, tools=None, stream=False) → response
       ├─► POST to http://localhost:11434/api/chat
       ├─► Request body:
       │    ├─ model: str
       │    ├─ messages: List[Dict]
       │    ├─ tools: List[Dict] (optional)
       │    └─ stream: bool
       ├─► Get response JSON
       ├─► make_serializable(response) → ensure JSON-compatible
       ├─► OllamaLogger.log_call(...) → SQLite insert
       └─► Return response dict with "message", "done", etc.


  QueryDecomposer.decompose(user_query, messages) → ExecutionPlan
       ├─► Create system_prompt (instructions for task decomposition)
       ├─► ollama_chat(model=self.model, messages=[system_prompt + user_input])
       ├─► Extract response content
       ├─► _extract_json(content) → dict of tasks
       ├─► For each task in dict:
       │    └─► Create Task(id, query, depends_on=[...])
       └─► Return ExecutionPlan(tasks)


  Task.run(db, model, tool_registry) → bool
       ├─► db.route_query(self.query) → suggested_tools
       ├─► DBConnection._load_tool_docs_map() → tool_docs
       ├─► Build selected_tools list from tool_docs
       ├─► ollama_chat(model, self.messages, tools=selected_tools)
       ├─► If tool_calls in response:
       │    ├─► For each tool_call:
       │    │    ├─► Extract tool_name and arguments
       │    │    ├─► result = tool_registry[tool_name](**arguments)
       │    │    ├─► Append result message to self.messages
       │    │    └─► Add tool_name to self.tools_used
       │    └─► ollama_chat again with context
       ├─► self.result = extracted_result
       ├─► self.status = "completed"
       └─► Return True if successful

╔═════╗    = Class or Component

╚═════╝

    └─► = Sub-function or related call

    ▼    = Next step or return
    ◄    = Feedback or return value

Data structures:
  List[T]   = Python list of type T
  Dict[K,V] = Python dictionary with key type K, value type V
  Tuple     = Fixed-size ordered collection
  set       = Unordered collection of unique items
  Optional  = May be None
  →         = Returns type arrow
